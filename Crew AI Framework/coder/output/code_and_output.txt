# Code to calculate the first 10,000 terms of the series
def calculate_series(n_terms):
    total = 0
    for i in range(n_terms):
        term = 1 / (2 * i + 1)  # Calculate the term 1/(2*i + 1)
        if i % 2 == 0:
            total += term  # Add for even index
        else:
            total -= term  # Subtract for odd index
    return total * 4  # Multiply the total by 4

# Calculate for the first 10,000 terms
result = calculate_series(10000)
print(result)
